Step 3: Create useGameStore (Pinia)
────────────────────────────────────
// stores/game.ts
import { ref } from "vue";
import { defineStore } from "pinia";

export const useGameStore = defineStore("game", () => {
  const gameId = ref("");
  const currentPlayer = ref("yellow");
  const pieces = ref<Record<string, string>>({}); // position → player color
  const validMoves = ref<string[]>([]);
  
  function movePiece(from: string, to: string) {
    const piece = pieces.value[from];
    if (piece) {
      delete pieces.value[from];
      pieces.value[to] = piece;
    }
  }
  
  function setValidMoves(positions: string[]) {
    validMoves.value = positions;
  }
  
  return { gameId, currentPlayer, pieces, validMoves, movePiece, setValidMoves };
});
